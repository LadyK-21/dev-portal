## All `type` and `color` variations

<table>
  <thead>
    <th>Color</th>
    <th>Type: filled</th>
    <th>Type: inverted</th>
    <th>Type: outlined</th>
  </thead>
  <tbody>
    {[
      'neutral',
      'neutral-dark-mode',
      'highlight',
      'success',
      'warning',
      'critical',
    ].map((color) => {
      return (
        <tr>
          <td>{color}</td>
          {['filled', 'inverted', 'outlined'].map((type) => {
            return (
              <td
                style={{
                  backgroundColor:
                    color === 'neutral-dark-mode' ? 'black' : 'transparent',
                  padding: 16,
                }}
              >
                {['small', 'medium', 'large'].map((size) => {
                  const isStatus =
                    color === 'success' ||
                    color === 'warning' ||
                    color === 'critical'
                  return (
                    <div
                      style={{
                        display: 'flex',
                        alignItems: 'center',
                        marginBottom: 8,
                        gap: 8,
                      }}
                    >
                      <Badge
                        color={color}
                        icon={<SwingsetTestIcon name="plus" />}
                        size={size}
                        text="Lorem ipsum"
                        type={type}
                      />
                      {/* To avoid the error */}
                      {!isStatus && (
                        <Badge
                          color={color}
                          size={size}
                          text="Lorem ipsum"
                          type={type}
                        />
                      )}
                      <Badge
                        ariaLabel="Lorem ipsum"
                        color={color}
                        size={size}
                        icon={<SwingsetTestIcon name="plus" />}
                        type={type}
                      />
                    </div>
                  )
                })}
              </td>
            )
          })}
        </tr>
      )
    })}
  </tbody>
</table>
